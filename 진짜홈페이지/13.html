import java.awt.*;
import java.awt.event.*;
import javax.swing.*;
class MyFrame extends JFrame {
 MyFrame() {
    setTitle("My Frame");
    setSize(600,400);
    setDefaultCloseOperation(EXIT_ON_CLOSE);
    JPanel pc = new JPanel();
    add(pc, BorderLayout.CENTER);
    
    JButton btn = new JButton("OK");
    btn.addActionListener(new ActionListener() {//익명의 내부 수신자
    @Override
     public void actionPerformed(ActionEvent e) {
        JOptionPane.showMessageDialog(MyFrame.this,"Button clicked"); //ActionListene 클래스 이므로 MyFrame.this를 한다.
     }
    });
    pc.add(btn);

setVisible(true);
}
public static void main(String[] args) {
new MyFrame();
}
}

Comppnent parent : main 창

int messageType : information 을 바꿈 <앞의 사진>

대화창은 parent 의 중앙에 나타 난다.
대화창이 사라지지 않으면 main 이 동작하지 않는다
