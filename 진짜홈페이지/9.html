public class RevLines extends JFrame implements ActionListener {

   private JTextArea ta;

      setTitle("Reverse Lines");
      setSize(600, 400);
      setDefaultCloseOperation(EXIT_ON_CLOSE);

      ta = new JTextArea();
      btns = new JButton[2];
      btns[0] = new JButton("OK");
      btns[1] = new JButton("Clear");
      for(JButton b : btns)
         b.addActionListener(this);
      add(new TextAreaButtons(ta, btns), BorderLayout.CENTER);

      setVisible(true);

   }

   public static void main(String[] args) {
      new RevLines();
   }

      JButton b = (JButton)e.getSource();

         ta.setText("");

      }
   }

import java.awt.BorderLayout;
import java.awt.Color;

import javax.swing.BorderFactory;
import javax.swing.JButton;
import javax.swing.JFrame;
import javax.swing.JPanel;
import javax.swing.JScrollPane;
import javax.swing.JTextArea;

public class TextAreaButtons extends JPanel {
   public TextAreaButtons(JTextArea ta, JButton[] btns) {
      setBorder(BorderFactory.createEmptyBorder(10, 10, 10, 10));
      setLayout(new BorderLayout(10, 10));
      JScrollPane pane = new JScrollPane(ta);
      pane.setBorder(BorderFactory.createLineBorder(Color.BLACK));
      add(pane, BorderLayout.CENTER);
      JPanel ps = new JPanel();
      for (int i = 0; i < btns.length; i++)
         ps.add(btns[i]);
      add(ps, BorderLayout.SOUTH);
   }

   public static void main(String[] args) {
      JFrame f = new JFrame("TextAreaButtons");
      f.setSize(400, 300);
      f.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
      JTextArea ta = new JTextArea();
      JButton[] btns = new JButton[2];
      btns[0] = new JButton("One");
      btns[1] = new JButton("Two");
      f.add(new TextAreaButtons(ta, btns), BorderLayout.CENTER);
      f.setVisible(true);
   }
}
