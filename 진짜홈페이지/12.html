 public void actionPerformed(AtcionEvent e){
    JButton b  = (JButton)e.getSource();
    if ( b == center){
     if(b.getText().equals("START")){
        b.setText("CENTER");
        up.setEnabled(true);
        down.setEnabled(true);
        left.setEnabled(true);
        right.setEnabled(true);
        }
      mp.center();
    }
    else if ( b==up)
        mp.up();
    else if ( b==down()
        mp.down();

    else if ( b==right()
