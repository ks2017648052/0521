2013년 중간고사 1번 문제

class Rect{
int width, height;

Rect(){
width = 1;
height = 1;
}
Rect(int width, int height){
this.width = width;
this.height = height;
}
int getWidth(){
return width;
}
int getHeight(){
return height;
}
int getArea(){
return width*height;
}
}

class Square extends Rect{
Square(){
width = 1;
height = 1;
}
Square(int width, int height){
this.width = width;
this.height = height;
}
}

class Test{
public static void main(String[] args){
Rect r1 = new Rect(3,5);
System.out.println(r1.getWidth() + " " + r1.getHeight() + " " + r1.getArea());
Square s1 = new Square(4,5);
System.out.println(s1.getWidth() + " " + s1.getHeight() + " " + s1.getArea());
}
}
